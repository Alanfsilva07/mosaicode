#################################
Installing and running the Harpia
#################################

**********
Installing
**********

.. code-block:: bash
   :linenos:

   # Go to into the Harpia folder that has been downloaded.
   cd harpia

   # Change branch
   git checkout newgui

   sudo python setup.py install

*******
Running
*******

.. code-block:: bash
   :linenos:

   harpia
